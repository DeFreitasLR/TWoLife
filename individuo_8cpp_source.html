130 int individuo::sorteia_acao()
131 {
132  vector<double> probscum;
133  double total = this->death+this->birth+this->move;
134  probscum.push_back(this->death/total);
135  probscum.push_back(probscum[0]+(this->birth/total));
136  probscum.push_back(probscum[1]+(this->move/total));
137  double evento;
138  evento = runif(0.0,1.0);
139 
140  int decisao;
141  for(unsigned int i=0; i<probscum.size()-1; i++)
142  {
143  if(probscum[i]>evento)//encontrar o primeiro maior valor
144  {
145  decisao = i;
146  return decisao;//retorna a decisao tomada para o mundo (0 = morte, 1 = nascer, 2 = andar)
147  break;
148  }
149  }
